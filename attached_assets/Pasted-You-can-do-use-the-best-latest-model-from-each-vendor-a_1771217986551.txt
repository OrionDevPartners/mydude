Drop-in code: model_resolver.py

import os, re, time, json
from dataclasses import dataclass
from typing import Optional, List, Dict, Tuple

CACHE_PATH = "model_cache.json"
CACHE_TTL_SECONDS = int(os.getenv("MODEL_CACHE_TTL_SECONDS", "21600"))  # 6 hours default

@dataclass(frozen=True)
class ResolvedModels:
    openai: str
    anthropic: str
    gemini: str
    grok: str

def _load_cache() -> dict:
    try:
        with open(CACHE_PATH, "r", encoding="utf-8") as f:
            return json.load(f)
    except Exception:
        return {}

def _save_cache(d: dict) -> None:
    with open(CACHE_PATH, "w", encoding="utf-8") as f:
        json.dump(d, f, ensure_ascii=False, indent=2)

def _is_fresh(cache: dict, key: str) -> bool:
    ts = cache.get(key, {}).get("ts", 0)
    return (time.time() - ts) < CACHE_TTL_SECONDS

def _semver_key(name: str) -> Tuple[int, int, int]:
    """
    Extracts x.y.z from strings like 'gpt-5.2' or 'claude-opus-4-6' (maps 4-6 -> 4.6.0).
    Non-matching => (0,0,0).
    """
    # gpt-5.2 or gpt-5.2.1
    m = re.search(r"(\d+)\.(\d+)(?:\.(\d+))?", name)
    if m:
        return (int(m.group(1)), int(m.group(2)), int(m.group(3) or 0))
    # claude-opus-4-6 style
    m2 = re.search(r"opus-(\d+)-(\d+)(?:-(\d+))?", name)
    if m2:
        return (int(m2.group(1)), int(m2.group(2)), int(m2.group(3) or 0))
    return (0, 0, 0)

def pick_latest_matching(model_ids: List[str], prefix_regex: str) -> Optional[str]:
    pat = re.compile(prefix_regex)
    candidates = [m for m in model_ids if pat.search(m)]
    if not candidates:
        return None
    candidates.sort(key=_semver_key, reverse=True)
    return candidates[0]

async def resolve_models(
    openai_list_models,
    anthropic_alias: Optional[str],
    gemini_list_models,
    grok_list_models
) -> ResolvedModels:
    """
    openai_list_models: async fn -> List[str]
    gemini_list_models: async fn -> List[str]
    grok_list_models: async fn -> List[str]
    anthropic_alias: if provided, use it (Anthropic aliases can track latest)  [oai_citation:8‡Claude Code](https://code.claude.com/docs/en/model-config?utm_source=chatgpt.com)
    """
    cache = _load_cache()

    # --- OpenAI ---
    if _is_fresh(cache, "openai"):
        openai_model = cache["openai"]["model"]
    else:
        o_models = await openai_list_models()
        # Policy: use latest gpt-5.x (today docs show gpt-5.2 exists)  [oai_citation:9‡OpenAI Developers](https://developers.openai.com/api/docs/models/gpt-5.2?utm_source=chatgpt.com)
        openai_model = pick_latest_matching(o_models, r"^gpt-5(\.|$)") or "gpt-5.2"
        cache["openai"] = {"ts": time.time(), "model": openai_model}

    # --- Anthropic ---
    # Best practice: use alias that always points to latest Opus if you want auto-rotate  [oai_citation:10‡Claude Code](https://code.claude.com/docs/en/model-config?utm_source=chatgpt.com)
    anthropic_model = anthropic_alias or os.getenv("ANTHROPIC_DEFAULT_OPUS_MODEL", "claude-opus-4-6")

    # --- Gemini ---
    if _is_fresh(cache, "gemini"):
        gemini_model = cache["gemini"]["model"]
    else:
        g_models = await gemini_list_models()
        # Policy: latest Gemini 3 Pro  [oai_citation:11‡Google AI for Developers](https://ai.google.dev/gemini-api/docs/models?utm_source=chatgpt.com)
        gemini_model = pick_latest_matching(g_models, r"gemini.*3.*pro") or "gemini-3-pro"
        cache["gemini"] = {"ts": time.time(), "model": gemini_model}

    # --- Grok ---
    if _is_fresh(cache, "grok"):
        grok_model = cache["grok"]["model"]
    else:
        x_models = await grok_list_models()
        # Policy: prefer Grok 4.1 Fast (agentic tool calling)  [oai_citation:12‡xAI](https://x.ai/news/grok-4-1-fast?utm_source=chatgpt.com)
        grok_model = pick_latest_matching(x_models, r"grok.*4\.1.*fast") or "grok-4.1-fast"
        cache["grok"] = {"ts": time.time(), "model": grok_model}

    _save_cache(cache)

    return ResolvedModels(
        openai=openai_model,
        anthropic=anthropic_model,
        gemini=gemini_model,
        grok=grok_model
    )
